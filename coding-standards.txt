    prefer 
    function launchRocket() {
        if (this.fuelCheck() && this.preflightChecks()) {
            this.startLaunchProcedure();
            this.launch();
        }
        else {
            this.notifyFailedChecks()
        }
    }

    function launchRocket() {
        //do fuel check
        ... etc. .. fuel check code 
        ...

        //if fuel check successful, do preflights checks
        ... etc. ... preflight check code written out 

        //if all that went well, start launch procedure
        .... etc.... you get the idea 
    }
